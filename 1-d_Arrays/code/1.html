1. Create a program to check wheter Array is Sorted or not.

        #include <stdio.h>
        #include <stdbool.h>
        void main()
        {
            int arr[] = {1, 2, 7, 8};
            int i = 0;
            bool set = true;
            for (i; i < 4; i++)
            {
              if (arr[i] > arr[i + 1])
              {
                 set = false;
                 break;
              }
            }
            if (set == false)
            {
              printf("Array is not Sorted");
            }
            else
            {
              printf("Array is Sorted");
            }
        }
